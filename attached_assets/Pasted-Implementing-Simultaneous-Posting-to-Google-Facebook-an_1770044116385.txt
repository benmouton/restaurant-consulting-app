Prerequisites (Do These First – 1–2 Hours)

Set Up Developer Accounts (free, but required for APIs):
Meta (Facebook & Instagram): Go to https://developers.facebook.com → Create App → Choose "Business" type → Get App ID & Secret. Link your Facebook Page and Instagram Business Account via Meta Business Suite (business.facebook.com).
Google Business Profile: Go to https://developers.google.com/my-business → Enable API in Google Cloud Console → Create OAuth credentials (Client ID & Secret). Note: Users need a verified Google Business location.

Paste this into your server.js or main backend file:
JavaScriptconst express = require('express');
const passport = require('passport');
const FacebookStrategy = require('passport-facebook').Strategy;
const GoogleStrategy = require('passport-google-oauth20').Strategy;
// Instagram uses same Meta API as Facebook for Business accounts

const app = express();

// Passport setup for Meta (FB/IG)
passport.use(new FacebookStrategy({
  clientID: process.env.META_APP_ID,
  clientSecret: process.env.META_APP_SECRET,
  callbackURL: `${process.env.APP_URL}/auth/meta/callback`,
  profileFields: ['id', 'displayName', 'emails']
}, (accessToken, refreshToken, profile, done) => {
  // Save accessToken to database for this user (e.g., user.metaToken = accessToken)
  return done(null, profile);
}));

// Google Business
passport.use(new GoogleStrategy({
  clientID: process.env.GOOGLE_CLIENT_ID,
  clientSecret: process.env.GOOGLE_CLIENT_SECRET,
  callbackURL: `${process.env.APP_URL}/auth/google/callback`
}, (accessToken, refreshToken, profile, done) => {
  // Save accessToken to database (user.googleToken = accessToken)
  return done(null, profile);
}));

// Routes to start linking
app.get('/auth/meta', passport.authenticate('facebook', { scope: ['pages_manage_posts', 'instagram_basic', 'business_management'] }));

app.get('/auth/google', passport.authenticate('google', { scope: ['https://www.googleapis.com/auth/business.manage'] }));

// Callback routes (after user approves)
app.get('/auth/meta/callback', passport.authenticate('facebook', { failureRedirect: '/social' }), (req, res) => {
  // Success: save token, redirect to social page
  res.redirect('/social?linked=meta');
});

app.get('/auth/google/callback', passport.authenticate('google', { failureRedirect: '/social' }), (req, res) => {
  res.redirect('/social?linked=google');
});

Backend route to handle posting (add to server.js):
JavaScriptconst axios = require('axios');

app.post('/api/post-social', async (req, res) => {
  const { content, platforms, userId } = req.body; // content from AI generator
  const user = await getUserFromDb(userId); // Fetch user with tokens

  try {
    if (platforms.includes('facebook') && user.metaToken) {
      // Post to FB Page (replace PAGE_ID with user's page ID from profile)
      await axios.post(`https://graph.facebook.com/PAGE_ID/feed`, {
        message: content,
        access_token: user.metaToken
      });
    }

    if (platforms.includes('instagram') && user.metaToken) {
      // Post to IG (needs container for media, but for text: use Graph API)
      await axios.post(`https://graph.facebook.com/IG_BUSINESS_ID/media`, {
        caption: content,
        access_token: user.metaToken
      });
    }

    if (platforms.includes('google') && user.googleToken) {
      // Post to Google Business (use Business Profile API)
      await axios.post('https://mybusiness.googleapis.com/v4/accounts/ACCOUNT_ID/locations/LOCATION_ID/localPosts', {
        summary: content,
      }, { headers: { Authorization: `Bearer ${user.googleToken}` } });
    }

    res.json({ success: true });
  } catch (err) {
    res.json({ error: err.message });
  }
});
